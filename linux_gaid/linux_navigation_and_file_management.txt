# cd          переход по директориям
cd (cd ~)     # переход в домашний директорий
cd~user       # переход в домашний директорий user
cd -          # переход в предыдущий директорий
cd ..         # переход в директорий выше
cd /          # переход в корневой директорий

# pwd выводит на экран полный абсолютный путь к текущей рабочей директории
pwd -F        # логический путь, показывает по которому вы перешли в директорию
pwd -L        # физический путь, показывает реальный физический путь, разрешая все символьные ссылки

# ls          просмотр содержимого каталога
ls -l         # показывает подробную информацию, вывод в виде списка
ls -a         # включает просмотр скрытых файлов
ls -A         # включает просмотр скрытых файлов, не отображает . и ..
ls -h         # выводит размер в удобно читаемом формате к,м,г
ls -F         # добавляет индикатор типа файла
ls --color	  # включить цветной вывод (обычно включен по умолчанию).
ls -t         # сортировать по времени последнего изменения (сначала новые)
ls -S         # сортировать по размеру (сначала большие)
ls -r         # обратный порядок сортировки (полезен в комбинациях с -tr и Sr)
ls -R         # рекурсивный обход (Показывает содержимое всех подкаталогов)
# Рекомендуеться создание алиаса: alias ll='ls -AlFh --color'

# cp команда для копирования файлов и каталогов
# cp <источник> <назначение>
cp -i         # запрашивает подтверждение перед перезаписью существующего файла
cp -r (cp -R) # копирует директории вместе со всем их содержимым
cp -p         # сохраняет права доступа, владельца, временные метки и другие атрибуты исходного файла
cp -a         # архивный режим (комбинация флагов -rp), сохраняет всю возможную информацию о файле и рекурсивно копирует директории
cp -v         # показывает подробную информацию о копируемых файлах
cp -n         # не перезаписывает существующие файлы
cp -f         # принудительное копирование
# работа с сылками
cp -l         # cоздание жесткой ссылки вместо копирования файлов (файлы с одним inode)
cp -s         # создание символьной ссылки вместо копирования файлов (ярлык)
# Рекомендуеться создание алиаса: alias cp='cp -i'

# mv команда для перемещения и переименования файлов и каталогов
# mv <источник> <назначение>
mv -i         # запрашивает подтверждение перед перезаписью существующего файла
mv -v         # показывает подробную информацию о перемещение файлов
mv -f         # принудительное выполнение
mv -b         # создать резервную копию, создает копию перезаписываемого файла с добавлением тильды (~) к его имени
# Рекомендуеться создание алиаса: alias mv='mv -i'

# rm команда для удаления файлов и директория
# rm <файл или директорий>
rm -f         # принудительное удаление
rm -r (rm -r) # рекурсивное удаление        
rm -i         # интерактивный режим
rm -v         # показывает подробную информацию во время выполнения
rm -d         # удаляет пустые директории
# Рекомендуеться создание алиаса: alias rm='rm -i'
